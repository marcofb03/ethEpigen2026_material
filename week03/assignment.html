Filter the peak that fall into a gene

library(ensembldb)
library(GenomicRanges)

#Save the original index on peaks
peaks$old_index <- seq_along(peaks)

#Get the Drosophila EnsDb and extract protein-coding genes
ensdb_dro <- ah[["AH119285"]]

prot_coding = FALSE

if(prot_coding) {

  dro_pcg <- genes(ensdb_dro, filter = ~ gene_biotype == "protein_coding")
  
  #Filter the peaks that fall inside a protein coding gene
  peaks_filtered <- subsetByOverlaps(peaks, dro_pcg)

} else {
  dro_all <- genes(ensdb_dro)
  
  #Filter the peaks that fall inside a gene
  peaks_filtered <- subsetByOverlaps(peaks, dro_all)
}
save(peaks_filtered, file = "peaks_filtered.RData")

PLOT

library(Gviz)
library(rtracklayer)

#LOADING FOR KNIT
options(ucscChromosomeNames = FALSE)
library(AnnotationHub)
ah <- AnnotationHub()
ensdb_dro <- ah[["AH119285"]]

#loading files
load("peaks_filtered.RData")
peaks <- rtracklayer::import("peaks/ctcf_peaks.narrowPeak")
treat_gr <- rtracklayer::import("peaks/ctcf_treat_pileup.bdg", format = "bedGraph")
lambda_gr <- rtracklayer::import("peaks/ctcf_control_lambda.bdg", format = "bedGraph")
xls <- read.table("peaks/ctcf_peaks.xls", header = TRUE, comment.char = "#")

#select the peak
i <- 1500
#selecting subset of peaks (Global vs Gene (protein_coding))
#my_peak <- peaks[i] 
my_peak <- peaks_filtered[i]

#Matching narrowPeak and XLS (handles the 1bp offset)
match_idx <- which(abs(xls$start - start(my_peak)) <= 1 & 
                   (xls$chr == as.character(seqnames(my_peak)) | 
                    xls$chr == gsub("chr", "", as.character(seqnames(my_peak)))))[1]

if(is.na(match_idx)) stop(paste("No match", i))

dati_xls <- xls[match_idx, ]

#extract values for labels
pileup_val   <- as.numeric(dati_xls$pileup)
fc_effective <- as.numeric(dati_xls$fold_enrichment)
bg_colored <- pileup_val / fc_effective

#track signal
strack <- DataTrack(range = treat_gr, 
                    name = "Signal", 
                    type = "histogram", 
                    fill = "gray70", 
                    col = "transparent")

#background line
ltrack <- DataTrack(range = lambda_gr, 
                    name = "Background", 
                    type = "polygon",
                    type = "l", 
                    col = "red", 
                    lwd = 3) 

#overlay
otrack <- OverlayTrack(trackList = list(strack, ltrack))

#peaks track
labels <- rep(" ", length(peaks)); labels[i] <- "TARGET"
colors <- rep("royalblue", length(peaks)); colors[i] <- "red"
ptrack <- AnnotationTrack(peaks, name = "Peaks", id = labels, showId = TRUE,
                          fill = colors, col = "transparent", just.group = "above")


#Add a Genome Axis (improves styling and readability)
atrack <- GenomeAxisTrack(add53 = TRUE, add35 = TRUE)

## ADDING SYMBOL LABEL INSTEAD OF TRANSCRIPT ID
#Extraction of data in the peak region
gene_data <- ensembldb::getGeneRegionTrackForGviz(ensdb_dro, 
                                        chromosome = as.character(seqnames(my_peak)), 
                                        start = start(my_peak) - 10000, 
                                        end = end(my_peak) + 10000)

#Manual substitution of Symbol instead of Transcript ID
gene_data$transcript <- ifelse(!is.na(gene_data$symbol) & gene_data$symbol != "", 
                               gene_data$symbol, 
                               gene_data$gene)

#track with symbol and not ID
gtrack <- GeneRegionTrack(gene_data,
                          name = "Genes",
                          transcriptAnnotation = "transcript", 
                          collapseTranscripts = "longest",     
                          fill = "#8282d2", 
                          col = "transparent")

plotTracks(list(atrack, otrack, ptrack, gtrack),
           chromosome = as.character(seqnames(my_peak)),
           from = start(my_peak) - 10000, 
           to = end(my_peak) + 10000,
           showId = TRUE,           
           geneSymbol = TRUE,      
           main = paste0("Peak n. ", peaks_filtered$old_index[i], " | Fold Change: ", round(fc_effective, 2), "\n",
                         "Local Background (Avg): ", round(bg_colored, 2), " reads"),
           cex.main = 0.9,
           fontface.main = 2,
           ylim = c(0, max(pileup_val * 1.5, 50)))
